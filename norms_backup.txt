import React, { useState, useEffect, ChangeEvent, MouseEvent } from 'react';
import { Plus, Search, Filter, Trash2, Edit, ChevronDown, ChevronUp, Info, X, ArrowLeft, Percent } from 'lucide-react';
import { Norm, Resource, ResourceType } from '../types';
import { motion, AnimatePresence } from 'motion/react';

type NewResource = {
  name: string;
  quantity: number;
  unit: string;
  resource_type: ResourceType;
  is_percentage: boolean;
  percentage_base: string;
};

export default function Norms() {
  // 1. Local State for Data
  const [norms, setNorms] = useState<Norm[]>([]);
  const [loading, setLoading] = useState(true);
  const [filter, setFilter] = useState<'ALL' | 'DOR' | 'DUDBC'>('ALL');
  const [search, setSearch] = useState('');
  const [isModalOpen, setIsModalOpen] = useState(false);
  const [editingNorm, setEditingNorm] = useState<Norm | null>(null);
  const [expandedNorm, setExpandedNorm] = useState<number | null>(null);

  // 2. FETCH data from Database on Load
  useEffect(() => {
    fetchNorms();
  }, []);

  const fetchNorms = async (): Promise<void> => {
    try {
      setLoading(true);
      const response = await fetch('http://localhost:5000/api/norms');
      if (!response.ok) {
        const errorText = await response.text();
        throw new Error(errorText || response.statusText);
      }
      const data: Norm[] = await response.json();
      setNorms(data);
    } catch (error) {
      console.error("Error fetching norms:", error);
      alert(`Failed to load norms from database. ${error instanceof Error ? error.message : ''}`);
    } finally {
      setLoading(false);
    }
  };

  // 3. SECURE DELETE from Database
  const handleDelete = async (id: number): Promise<void> => {
    if (!confirm('Are you sure you want to delete this norm?')) return;
    
    try {
      const response = await fetch(`http://localhost:5000/api/norms?id=${id}`, { method: 'DELETE' });
      if (response.ok) {
        setNorms((prev: Norm[]) => prev.filter((n: Norm) => n.id !== id));
      }
    } catch (error) {
      alert("Error deleting norm.");
    }
  };

  // 4. SECURE SAVE (Create or Update) to Database
  const handleSave = async (normData: Norm): Promise<void> => {
    const method = editingNorm ? 'PUT' : 'POST';
    
    try {
      const response = await fetch('http://localhost:5000/api/norms', {
        method,
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(normData),
      });

      if (response.ok) {
        // Refresh the list from DB to ensure data integrity
        await fetchNorms();
        setIsModalOpen(false);
      }
    } catch (error) {
      alert("Error saving norm to database.");
    }
  };

  const filteredNorms = norms.filter((n: Norm) => {
    const matchesFilter = filter === 'ALL' || n.type === filter;
    const matchesSearch = n.description.toLowerCase().includes(search.toLowerCase()) || 
                         n.ref_ss?.toLowerCase().includes(search.toLowerCase());
    return matchesFilter && matchesSearch;
  });

  if (loading) return <div className="p-10 text-center font-bold">Loading Norms...</div>;

  return (
    <div className="min-h-screen bg-[#F8F9FA] pb-24">
      {/* Header */}
      <header className="sticky top-0 z-30 bg-white/80 backdrop-blur-md border-b border-black/5 px-6 py-4">
        <div className="flex items-center justify-between max-w-md mx-auto">
          <h1 className="text-xl font-bold tracking-tight italic">QuantBreak Norms</h1>
          <button 
            onClick={() => { setEditingNorm(null); setIsModalOpen(true); }}
            className="w-10 h-10 bg-[#141414] text-white rounded-xl flex items-center justify-center shadow-lg shadow-black/10 active:scale-95 transition-all"
          >
            <Plus size={20} />
          </button>
        </div>
      </header>

      <main className="max-w-md mx-auto p-4 space-y-4">
        {/* Search & Filter */}
        <div className="space-y-3">
          <div className="relative">
            <Search className="absolute left-4 top-1/2 -translate-y-1/2 text-black/30" size={18} />
            <input 
              type="text" 
              placeholder="Search library..."
              className="w-full pl-12 pr-4 py-3 bg-white border border-black/5 rounded-2xl focus:outline-none focus:ring-2 focus:ring-emerald-500/20 transition-all text-sm"
              value={search}
              onChange={(e: ChangeEvent<HTMLInputElement>) => setSearch(e.target.value)}
            />
          </div>
          <div className="flex bg-white p-1 rounded-2xl border border-black/5 overflow-x-auto no-scrollbar">
            {(['ALL', 'DOR', 'DUDBC'] as const).map(type => (
              <button
                key={type}
                onClick={() => setFilter(type)}
                className={`flex-1 px-4 py-2 rounded-xl text-[10px] font-bold uppercase tracking-widest transition-all whitespace-nowrap ${
                  filter === type ? 'bg-[#141414] text-white shadow-md' : 'text-black/40 hover:text-black/60'
                }`}
              >
                {type}
              </button>
            ))}
          </div>
        </div>

        {/* Norm Cards */}
        <div className="space-y-3">
          {filteredNorms.map(norm => (
            <motion.div 
              layout
              key={norm.id}
              className="bg-white rounded-3xl border border-black/5 shadow-sm overflow-hidden"
            >
              <div 
                onClick={() => setExpandedNorm(expandedNorm === norm.id ? null : norm.id)}
                className="p-5 space-y-3 active:bg-black/5 transition-colors"
              >
                <div className="flex items-start justify-between">
                  <div className="flex items-center gap-2">
                    <span className={`px-2 py-0.5 rounded-md text-[9px] font-bold uppercase ${
                      norm.type === 'DOR' ? 'bg-blue-100 text-blue-700' : 'bg-purple-100 text-purple-700'
                    }`}>
                      {norm.type}
                    </span>
                    <span className="text-[10px] font-mono text-black/30">{norm.ref_ss}</span>
                  </div>
                  <div className="flex items-center gap-1" onClick={(e: MouseEvent<HTMLDivElement>) => e.stopPropagation()}>
                    <button 
                      onClick={() => { setEditingNorm(norm); setIsModalOpen(true); }}
                      className="p-2 text-black/30 hover:text-black transition-colors"
                    >
                      <Edit size={14} />
                    </button>
                    <button 
                      onClick={() => handleDelete(norm.id)}
                      className="p-2 text-red-300 hover:text-red-500 transition-colors"
                    >
                      <Trash2 size={14} />
                    </button>
                  </div>
                </div>
                
                <div>
                  <h3 className="font-bold text-sm leading-tight">{norm.description}</h3>
                  <p className="text-[10px] text-black/40 mt-1">Unit: {norm.unit} • Basis: {norm.basis_quantity}</p>
                </div>

                <div className="flex items-center justify-between pt-2 border-t border-black/5">
                  <span className="text-[10px] font-bold text-black/20 uppercase tracking-widest">
                    {norm.resources.length} Resources
                  </span>
                  {expandedNorm === norm.id ? <ChevronUp size={16} className="text-black/20" /> : <ChevronDown size={16} className="text-black/20" />}
                </div>
              </div>

              <AnimatePresence>
                {expandedNorm === norm.id && (
                  <motion.div
                    initial={{ height: 0, opacity: 0 }}
                    animate={{ height: 'auto', opacity: 1 }}
                    exit={{ height: 0, opacity: 0 }}
                    className="bg-[#F5F5F0]/50 border-t border-black/5 p-5 space-y-4"
                  >
                    <div className="space-y-4">
                      <ResourceGroup title="Labour" resources={norm.resources.filter(r => r.resource_type === 'Labour')} color="blue" />
                      <ResourceGroup title="Material" resources={norm.resources.filter(r => r.resource_type === 'Material')} color="emerald" />
                      <ResourceGroup title="Equipment" resources={norm.resources.filter(r => r.resource_type === 'Equipment')} color="orange" />
                    </div>
                  </motion.div>
                )}
              </AnimatePresence>
            </motion.div>
          ))}
        </div>
      </main>

      {isModalOpen && (
        <NormModal 
          norm={editingNorm} 
          onClose={() => setIsModalOpen(false)} 
          onSave={handleSave} 
        />
      )}
    </div>
  );
}

// --- SUB-COMPONENTS (Keep these below main export) ---

function ResourceGroup({ title, resources, color }: { title: string, resources: Resource[], color: string }) {
  if (resources.length === 0) return null;
  const colorMap: Record<string, string> = {
    blue: 'text-blue-600/60',
    emerald: 'text-emerald-600/60',
    orange: 'text-orange-600/60'
  };
  return (
    <div className="space-y-2">
      <h4 className={`text-[9px] font-bold uppercase tracking-widest ${colorMap[color]}`}>{title}</h4>
      <div className="space-y-1">
        {resources.map((r, i) => (
          <div key={i} className="flex justify-between items-center text-xs">
            <span className="text-black/60">{r.name}</span>
            {r.is_percentage ? (
              <span className="font-mono font-bold text-purple-600">
                {r.quantity}% <span className="text-[9px] opacity-40">of {r.percentage_base}</span>
              </span>
            ) : (
              <span className="font-mono font-bold">
                {r.quantity} <span className="text-[9px] opacity-40">{r.unit || ''}</span>
              </span>
            )}
          </div>
        ))}
      </div>
    </div>
  );
}

type NormModalProps = {
  norm: Norm | null;
  onClose: () => void;
  onSave: (norm: Norm) => void;
};

function NormModal({ norm, onClose, onSave }: NormModalProps) {
  const [formData, setFormData] = useState<Norm>(norm || {
    id: 0,
    type: 'DOR',
    description: '',
    unit: '',
    basis_quantity: 1,
    ref_ss: '',
    resources: []
  });

  const [newResource, setNewResource] = useState<NewResource>({
    resource_type: 'Labour',
    name: '',
    unit: '',
    quantity: 0,
    is_percentage: false,
    percentage_base: 'Labour'
  });

  const [resourceTypeMode, setResourceTypeMode] = useState<'fixed' | 'percentage'>('fixed');

  const addResource = (): void => {
    if (resourceTypeMode === 'percentage') {
      if (!newResource.name || !newResource.quantity || newResource.quantity <= 0) return;
      setFormData({
        ...formData,
        resources: [...formData.resources, { 
          name: newResource.name,
          quantity: newResource.quantity,
          resource_type: newResource.resource_type,
          is_percentage: true,
          percentage_base: newResource.percentage_base,
          unit: ''
        }]
      });
    } else {
      if (!newResource.name || !newResource.quantity || newResource.quantity <= 0 || !newResource.unit) return;
      setFormData({
        ...formData,
        resources: [...formData.resources, { 
          name: newResource.name,
          quantity: newResource.quantity,
          unit: newResource.unit,
          resource_type: newResource.resource_type,
          is_percentage: false
        }]
      });
    }
    setNewResource({ ...newResource, name: '', unit: '', quantity: 0, is_percentage: false });
  };

  const removeResource = (index: number): void => {
    setFormData({ ...formData, resources: formData.resources.filter((_, i) => i !== index) });
  };

  return (
    <div className="fixed inset-0 z-[100] bg-white flex flex-col">
      <header className="p-6 border-b border-black/5 flex items-center justify-between bg-white sticky top-0">
        <button onClick={onClose} className="p-2 hover:bg-black/5 rounded-full transition-colors">
          <ArrowLeft size={20} />
        </button>
        <h2 className="text-lg font-bold">{norm ? 'Edit Norm' : 'New Norm'}</h2>
        <button 
          onClick={() => onSave(formData)}
          className="bg-[#141414] text-white px-6 py-2 rounded-xl text-xs font-bold uppercase tracking-widest shadow-lg shadow-black/10 active:scale-95 transition-all"
        >
          Save to DB
        </button>
      </header>
      
      <div className="flex-1 overflow-y-auto p-6 space-y-8">
        {/* Form contents same as your original, just ensure Save button triggers onSave(formData) */}
        <div className="space-y-4">
          <h3 className="text-xs font-bold uppercase tracking-widest text-black/40 px-1">Basic Information</h3>
          <div className="space-y-4 bg-[#F5F5F0] p-6 rounded-3xl">
            <div className="space-y-2">
              <label className="text-[10px] font-bold uppercase tracking-widest text-black/40">Type</label>
              <div className="flex bg-white p-1 rounded-xl border border-black/5">
                {(['DOR', 'DUDBC'] as const).map(t => (
                  <button key={t} onClick={() => setFormData({ ...formData, type: t })}
                    className={`flex-1 py-2 rounded-lg text-[10px] font-bold transition-all ${formData.type === t ? 'bg-[#141414] text-white' : 'text-black/40'}`}>
                    {t}
                  </button>
                ))}
              </div>
            </div>
            <textarea className="w-full p-4 bg-white rounded-xl text-sm min-h-[100px]" placeholder="Description..." value={formData.description} onChange={(e: ChangeEvent<HTMLTextAreaElement>) => setFormData({ ...formData, description: e.target.value })} />
            <div className="grid grid-cols-2 gap-4">
              <input type="text" className="w-full p-3 bg-white rounded-xl text-sm" placeholder="Unit (m³)" value={formData.unit} onChange={(e: ChangeEvent<HTMLInputElement>) => setFormData({ ...formData, unit: e.target.value })} />
              <input type="number" className="w-full p-3 bg-white rounded-xl text-sm" placeholder="Basis Qty" value={formData.basis_quantity} onChange={(e: ChangeEvent<HTMLInputElement>) => setFormData({ ...formData, basis_quantity: parseFloat(e.target.value) || 1 })} />
            </div>
          </div>
        </div>

        {/* Resource Section */}
        <div className="space-y-4">
          <h3 className="text-xs font-bold uppercase tracking-widest text-black/40 px-1">Resource Breakdown</h3>
          <div className={`p-6 rounded-3xl border space-y-4 ${resourceTypeMode === 'percentage' ? 'bg-purple-50' : 'bg-emerald-50'}`}>
             <div className="grid grid-cols-2 gap-3">
              <select className="p-3 bg-white rounded-xl text-xs font-bold" value={newResource.resource_type} onChange={(e: ChangeEvent<HTMLSelectElement>) => setNewResource({ ...newResource, resource_type: e.target.value as ResourceType })}>
                <option value="Labour">Labour</option>
                <option value="Material">Material</option>
                <option value="Equipment">Equipment</option>
              </select>
              {resourceTypeMode === 'fixed' ? (
                <input type="text" placeholder="Unit" className="p-3 bg-white rounded-xl text-xs" value={newResource.unit || ''} onChange={(e: ChangeEvent<HTMLInputElement>) => setNewResource({ ...newResource, unit: e.target.value })} />
              ) : (
                <select className="p-3 bg-white rounded-xl text-xs font-bold" value={newResource.percentage_base} onChange={(e: ChangeEvent<HTMLSelectElement>) => setNewResource({ ...newResource, percentage_base: e.target.value })}>
                  <option value="Labour">% of Labour</option>
                  <option value="Material">% of Material</option>
                  <option value="Total">% of Total</option>
                </select>
              )}
            </div>
            <input type="text" placeholder="Resource Name" className="w-full p-3 bg-white rounded-xl text-xs" value={newResource.name} onChange={(e: ChangeEvent<HTMLInputElement>) => setNewResource({ ...newResource, name: e.target.value })} />
            <div className="flex gap-3">
               <input type="number" placeholder="Qty" className="flex-1 p-3 bg-white rounded-xl text-xs" value={newResource.quantity} onChange={(e: ChangeEvent<HTMLInputElement>) => setNewResource({ ...newResource, quantity: parseFloat(e.target.value) || 0 })} />
               <button onClick={addResource} className="bg-black text-white px-6 py-3 rounded-xl font-bold text-xs">Add</button>
            </div>
          </div>

          <div className="space-y-2">
            {formData.resources.map((res: Resource, idx: number) => (
              <div key={idx} className="bg-white p-4 rounded-2xl border border-black/5 flex items-center justify-between">
                <div>
                  <p className="text-xs font-bold">{res.name}</p>
                  <p className="text-[10px] text-black/30 uppercase">{res.resource_type}</p>
                </div>
                <div className="flex items-center gap-4">
                  <span className="text-sm font-bold">{res.quantity} {res.is_percentage ? '%' : res.unit}</span>
                  <button onClick={() => removeResource(idx)} className="text-red-300"><Trash2 size={14} /></button>
                </div>
              </div>
            ))}
          </div>
        </div>
      </div>
    </div>
  );
}
